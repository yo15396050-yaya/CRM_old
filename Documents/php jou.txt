    // Enregistrement
    public function store(Request $request)
    {
        $request->validate([
            'code_journal' => 'required|unique:journal_tresorerie',
            'intitule' => 'required',
            'compte_de_contrepartie' => 'required',
        ]);


         if (substr($request->compte_de_contrepartie, 0, 1) !== '5') {
        return redirect()->back()
            ->withInput()
            ->withErrors(['compte_de_contrepartie' => "La particularité des journaux de trésorerie, c’est qu’à leur paramétrage, il faut leur un compte général de trésorerie de la classe 5, qui est renommé du nom du type de trésorerie."]);
    }

        JournalTresorerie::create([
            'code_journal' => $request->code_journal,
            'intitule' => $request->intitule,
            'traitement_analytique' => $request->traitement_analytique,
            'compte_de_contrepartie' => $request->compte_de_contrepartie,
            'rapprochement_sur' => $request->rapprochement_sur,

            'user_id' => Auth::id(),
            'company_id' => $request->company_id ?? null,
        ]);

        return redirect()->route('journal_tresorerie.index')
            ->with('success', 'Journal créé avec succès.');
    }

    public function show($id)
    {
        $journal = JournalTresorerie::findOrFail($id);
        return view('journal_tresorerie.show', compact('journal'));
    }

    // Formulaire édit
    public function edit($id)
    {
        $journal = JournalTresorerie::findOrFail($id);
        $comptesClasse5 = PlanComptable::where('numero_de_compte', 'like', '5%')->get();

        return view('journal_tresorerie.edit', compact('journal','comptesClasse5'));
    }




    // Mise à jour
    public function update(Request $request, $id)
    {
        $journal = JournalTresorerie::findOrFail($id);

        $request->validate([
            'code_journal' => 'required|unique:journal_tresorerie,code_journal,' . $journal->id,
            'intitule' => 'required',
        ]);

        $journal->update($request->all());

        return redirect()->route('journal_tresorerie.index')
            ->with('success', 'Journal mis à jour avec succès.');
    }

    // Suppression
    public function destroy($id)
    {
        $journal = JournalTresorerie::findOrFail($id);
        $journal->delete();

        return redirect()->route('journal_tresorerie.index')
            ->with('success', 'Journal supprimé avec succès.');
    }
